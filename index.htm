<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/12.8.0/firebase-app.js";
import { getDatabase, ref, set, onValue } from "https://www.gstatic.com/firebasejs/12.8.0/firebase-database.js";

messagingSenderId: "893127699685",

measurementId: "G-CKPYDX8JBV",
databaseURL: "https://k-jg-94ba5-default-rtdb.firebaseio.com" // 請確認這行，通常是 https://你的projectId-default-rtdb.firebaseio.com
};

// 初始化 Firebase
const app = initializeApp(firebaseConfig);
const db_cloud = getDatabase(app);

const products = ["翅", "腿", "麻吉燒", "腿排", "心", "尾", "腱", "米血", "甜", "豆包", "豬肉串", "熱狗", "香菇", "燻培鳳梨", "豆干", "培根卷", "雞肉串", "米腸", "雞皮", "德式香腸", "雞肉卷", "百頁", "黃金雞腿", "豬肉香腸", "肥腸", "雞軟骨", "烏龜排", "貢丸", "牛叻條串", "雪花豬", "節瓜", "中節翅", "雞松阪", "松阪豬", "蔥肉捲", "杏鮑菇", "皎白筍", "四季豆", "娃娃菜"];

let currentType = '生貨';
let db = { '生貨': {}, '熟貨': {} };
let historyDb = { '生貨': {}, '熟貨': {} };

// 初始化本地結構
products.forEach(p => {
db['生貨'][p] = { last: '', add: '', cur: '', locked: true };
db['熟貨'][p] = { last: '', add: '', cur: '', locked: true };
});

// --- 核心功能：雲端讀取 ---
function loadFromFirebase() {
const loc = document.getElementById('loc').value;
const dataRef = ref(db_cloud, 'inventory/' + loc);

onValue(dataRef, (snapshot) => {
const data = snapshot.val();
if (data) {
db = data.now || db;
historyDb = data.hist || historyDb;
render();
}
});
}

// --- 核心功能：雲端更新 ---
window.updateVal = function(p, field, val) {
db[currentType][p][field] = val;
const loc = document.getElementById('loc').value;
set(ref(db_cloud, 'inventory/' + loc + '/now'), db);
};

// --- 核心功能：轉載至明日 ---
window.prepareNextDay = function() {
if (!confirm("確定要【送單】嗎？今日現存將轉為明日昨存並存入歷史。")) return;
const loc = document.getElementById('loc').value;

// 先存歷史
set(ref(db_cloud, 'inventory/' + loc + '/hist'), db);

// 轉換數據
['生貨', '熟貨'].forEach(type => {
products.forEach(p => {
db[type][p].last = db[type][p].cur || 0;
db[type][p].add = '';
db[type][p].cur = '';
db[type][p].locked = true;
});
});
set(ref(db_cloud, 'inventory/' + loc + '/now'), db);
alert("雲端同步完成！");
};

// --- 其他介面功能 (需掛在 window 下讓 HTML 鈕能按到) ---
window.switchType = function(type) {
currentType = type;
document.getElementById('btn-raw').classList.toggle('active', type === '生貨');
document.getElementById('btn-cooked').classList.toggle('active', type === '熟貨');
render();
};

window.showHistory = function(p) {
const hData = historyDb[currentType][p];
document.getElementById('modal-item-name').innerText = p + " (上期紀錄)";
const modalBody = document.getElementById('modal-body');

if (hData && hData.last !== '') {
const hSold = (Number(hData.last)||0) + (Number(hData.add)||0) - (Number(hData.cur)||0);
modalBody.innerHTML = `
<div class="history-data">
<div>昨存：<b>${hData.last}</b></div>
<div>進貨：<b>${hData.add || '-'}</b></div>
<div>現存：<b>${hData.cur}</b></div>
<div>銷貨：<b style="color:red">${hSold}</b></div>
</div>`;
} else {
modalBody.innerHTML = "<p>尚未有上期紀錄</p>";
}
document.getElementById('history-modal').style.display = 'flex';
};

window.closeModal = function() {
document.getElementById('history-modal').style.display = 'none';
};

window.render = function() {
const list = document.getElementById('card-list');
list.innerHTML = products.map((p) => {
const data = db[currentType][p];
const lastVal = Number(data.last) || 0;
const addVal = (currentType === '生貨') ? (Number(data.add) || 0) : 0;
const curVal = Number(data.cur) || 0;
const sold = lastVal + addVal - curVal;

return `
<div class="card">
<div class="card-header">
<span class="item-name">${p}</span>
<button class="history-btn" onclick="showHistory('${p}')"> 歷史</button>
</div>
<div class="input-grid">
<div class="input-box">
<label>昨存</label>
<input type="number" class="qty-input ${data.locked ? 'qty-locked' : ''}"
${data.locked ? 'readonly' : ''} value="${data.last}"
onchange="updateVal('${p}', 'last', this.value)">
<button style="font-size:9px; margin-top:2px;" onclick="toggleLock('${p}')">鎖定/解除</button>
</div>
${currentType === '生貨' ? `
<div class="input-box">
<label>今日進貨</label>
<input type="number" class="qty-input" value="${data.add}"
onchange="updateVal('${p}', 'add', this.value)">
</div>` : ''}
<div class="input-box" style="${currentType === '熟貨' ? 'grid-column: span 2' : ''}">
<label>今日現存</label>
<input type="number" class="qty-input" value="${data.cur}"
onchange="updateVal('${p}', 'cur', this.value)">
</div>
</div>
<div style="text-align:right; margin-top:8px; font-weight:bold; color:red;">銷貨：${sold}</div>
</div>`;
}).join('');
};

window.toggleLock = function(p) {
db[currentType][p].locked = !db[currentType][p].locked;
const loc = document.getElementById('loc').value;
set(ref(db_cloud, 'inventory/' + loc + '/now'), db);
};

// 初始載入
document.getElementById('date').valueAsDate = new Date();
document.getElementById('loc').addEventListener('change', loadFromFirebase);
loadFromFirebase();
</script>
